async def capture_tokens_all_types(browser_context, pages):
    """
    Capture authentication tokens for all 3 user types.
    Logs in as each type and extracts tokens.
    """
    tokens_by_type = {
        "Member": None,
        "ExternalStaff": None,
        "NationalOfficeStaff": None
    }
    
    test_accounts = [
        {"email": "member@test.com", "password": "...", "type": "Member"},
        {"email": "staff@test.com", "password": "...", "type": "ExternalStaff"},
        {"email": "admin@test.com", "password": "...", "type": "NationalOfficeStaff"},
    ]
    
    for account in test_accounts:
        page = await browser_context.new_page()
        
        # Navigate to login
        await page.goto("https://league.poolplayers.com/login")
        
        # Enter credentials
        await page.fill("input[type='email']", account["email"])
        await page.fill("input[type='password']", account["password"])
        await page.click("button[type='submit']")
        
        # Wait for redirect to dashboard
        await page.wait_for_url("**/dashboard")
        
        # Extract tokens from localStorage
        tokens = await page.evaluate("""
            () => ({
                accessToken: localStorage.getItem('accessToken'),
                refreshToken: localStorage.getItem('refreshToken')
            })
        """)
        
        tokens_by_type[account["type"]] = tokens
        await page.close()
    
    return tokens_by_type
